Code:
try {
  const response = $input.first().json.choices[0].message.content;
  
  // Remove markdown code blocks if present
  let cleaned = response.trim();
  if (cleaned.startsWith('```json')) {
    cleaned = cleaned.replace(/```json\n?/g, '').replace(/```\n?$/g, '');
  }
  if (cleaned.startsWith('```')) {
    cleaned = cleaned.replace(/```\n?/g, '').replace(/```\n?$/g, '');
  }
  
  // Parse JSON
  const newsletter = JSON.parse(cleaned);
  
  // Validate required fields
  if (!newsletter.meta || !newsletter.sections || !newsletter.quote) {
    throw new Error('Missing required fields in newsletter JSON');
  }
  
  // Check word count
  const totalWords = newsletter.sections.reduce((sum, s) => sum + (s.word_count || 0), 0);
  if (totalWords < 4000) {
    throw new Error(`Word count too low: ${totalWords} words (min 4000)`);
  }
  
  return {
    json: {
      success: true,
      newsletter: newsletter,
      validation: {
        word_count: totalWords,
        sections_count: newsletter.sections.length,
        has_quote: !!newsletter.quote,
        has_resources: newsletter.resources && newsletter.resources.length > 0
      }
    }
  };
  
} catch (error) {
  return {
    json: {
      success: false,
      error: error.message,
      raw_response: $input.first().json.choices[0].message.content
    }
  };
}
